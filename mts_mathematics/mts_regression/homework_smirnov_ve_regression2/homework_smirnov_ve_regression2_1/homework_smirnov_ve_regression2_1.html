# Количество итераций
num_iterations = 10_000

# Словари для хранения результатов
inside_ci_counts = {'beta1': 0, 'beta2': 0, 'beta3': 0}

# Истинные коэффициенты модели
true_betas = {'beta1': dist_params['beta1'], 
              'beta2': dist_params['beta2'], 
              'beta3': dist_params['beta3']}

for _ in range(num_iterations):
    data = gen_data(y_type='multivariate', params=dist_params, seed=None)
    _, model = train_model(data, target='y', feature_names=['x1', 'x2', 'x3'])

    conf_intervals = model.conf_int()

    # Проверяем попадание истинных значений в доверительные интервалы
    for i, beta in enumerate(['beta1', 'beta2', 'beta3']):
        lower, upper = conf_intervals.iloc[i + 1]
        if lower <= true_betas[beta] <= upper:
            inside_ci_counts[beta] += 1

# Рассчитываем доли попаданий
ci_coverage = {beta: inside_ci_counts[beta] / num_iterations for beta in inside_ci_counts}

print(f"Доля попаданий β1 в доверительный интервал: {ci_coverage['beta1']:.3f}")
print(f"Доля попаданий β2 в доверительный интервал: {ci_coverage['beta2']:.3f}")
print(f"Доля попаданий β3 в доверительный интервал: {ci_coverage['beta3']:.3f}")
